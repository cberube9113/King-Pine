Helps with username lookups

  {id: 0,

    idlookup
  






Look up a user's ID

        		
        




        		
      

      

        


  ¶




Params 

  username string Username of user 

Returns 

  integer ID of user specified by username
   

        		
        

exports.idlookup = function(username){
  var c = -1;
  for(var i = 0; i < iddb.length; i++){
    if(iddb[i].uname == username){
      c = iddb[i].id;
      break;
    }
  }
  return c;
}
