    idlookup
  






ID Lookup Function
Resolves uid value given username.

        		
        

exports.idlookup = function(username){
  var id = -1;
  for(var i = 0; i < userdb.length; i++){
    if(userdb[i].username === username){
      id = i;
      break;
    }
  }
  
  return id;
}

    usernamelookup
  






Username Lookup Function
Resolves username given uid value.

        		
        

exports.usernamelookup = function(uid){
  var username = null;
  for(var i = 0; i < userdb.length; i++){
    if(userdb[i].uid == uid){
      username = userdb[i].username;
      break;
    }
  }
}

exports.login = function(username, password, cb){
  var len = userdb.length;
  for (var i = 0; i < len; i++){
    var u = userdb[i];
    if (u.username === username){
      if(u.password === password){
        cb(undefined, u);
      }
      else{
        cb('Password is incorrect.', undefined);
      }
      return;
    }
  }
  cb('User not found!', undefined);
};
